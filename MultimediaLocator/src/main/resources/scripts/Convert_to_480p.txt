on run arguments	set inputFolder to (first item of arguments)	set inputFiles to (list folder (inputFolder) without invisibles)	tell application "Finder" to set outputFolder to (second item of arguments) as «class furl»		repeat with documentName in inputFiles		set currentItem to (inputFolder & documentName) as POSIX file		tell application "QuickTime Player"			open currentItem			set bounds of window 1 to {-3000, 0, 100, 100}			if window 2 exists then				set bounds of window 2 to {6000, 3000, 0, 0}			end if			set movieDuration to the duration of the front document			export document 1 in file ((outputFolder as text) & documentName) using settings preset "480p"			close document 1 saving no		end tell		delay movieDuration * 0.5	end repeat		tell application "QuickTime Player"		quit	end tellend run
